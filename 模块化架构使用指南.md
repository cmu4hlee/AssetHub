## 1. 快速开始

### 1.1 应用启动时初始化模块

在 `server.js` 中添加模块初始化：

```javascript
const { initializeModules } = require('./init-modules');

// 在数据库连接之后、路由注册之前
async function startServer() {
  // ... 数据库初始化 ...
  
  // 初始化模块注册中心
  await initializeModules();
  
  // ... 路由注册 ...
}
```

### 1.2 使用模块

```javascript
const moduleRegistry = require('./config/module-registry');
const { asyncHandler, createModuleError } = require('./middleware/module-error-handler');

// 在路由中使用
router.get('/assets/:id', asyncHandler(async (req, res) => {
  const assetModule = moduleRegistry.get('asset');
  if (!assetModule) {
    throw createModuleError('DEPENDENCY_ERROR', '资产模块未注册');
  }

  const asset = await assetModule.findById(req.params.id, req.tenantId);
  if (!asset) {
    throw createModuleError('NOT_FOUND', '资产不存在');
  }

  res.json({ success: true, data: asset });
}));
```

## 2. 模块注册

### 2.1 注册新模块

```javascript
const moduleRegistry = require('./config/module-registry');
const MyModuleAdapter = require('./adapters/my-module-adapter');

moduleRegistry.register('my-module', new MyModuleAdapter(), {
  config: {
    // 模块配置
    timeout: 30,
    enabled: true,
  },
  dependencies: ['asset', 'quality-control'], // 依赖的其他模块
  enabled: true,
});
```

### 2.2 模块适配器标准接口

每个模块适配器应实现以下接口：

```javascript
class MyModuleAdapter {
  // 初始化
  async initialize(config) { }
  
  // 标准 CRUD 接口
  async findById(id, tenantId) { }
  async findAll(filters, tenantId, pagination) { }
  async create(data, tenantId, createdBy) { }
  async update(id, data, tenantId, updatedBy) { }
  async delete(id, tenantId) { }
  async count(filters, tenantId) { }
  
  // 生命周期
  async start() { }
  async stop() { }
  async destroy() { }
}
```

## 3. 模块间通信

### 3.1 发布事件

```javascript
moduleRegistry.emitEvent('my-module', 'event-name', {
  data: 'value',
});
```

### 3.2 监听事件

```javascript
moduleRegistry.onEvent('my-module', 'event-name', (payload) => {
  console.log('收到事件:', payload);
  // 处理事件
});
```

### 3.3 移除事件监听

```javascript
const handler = (payload) => { /* ... */ };
moduleRegistry.onEvent('my-module', 'event-name', handler);
// 稍后移除
moduleRegistry.offEvent('my-module', 'event-name', handler);
```

### 3.4 常用事件示例

| 模块 | 事件 | 触发时机 |
|------|------|---------|
| asset | created | 资产创建 |
| asset | updated | 资产更新 |
| asset | deleted | 资产删除 |
| asset | status-changed | 资产状态变更 |
| quality-control | created | 质控记录创建 |
| quality-control | updated | 质控记录更新 |
| quality-control | inspection-required | 需要质控检查 |

## 4. 配置管理

### 4.1 注册配置

```javascript
const moduleConfigManager = require('./config/module-config-manager');

moduleConfigManager.register('my-module', {
  timeout: 30,
  enabled: true,
  retry: { max: 3, delay: 1000 },
}, {
  timeout: { required: true, type: 'number', validate: v => v > 0 },
  enabled: { required: true, type: 'boolean' },
});
```

### 4.2 加载配置

```javascript
await moduleConfigManager.load('my-module');
```

### 4.3 获取配置值

```javascript
// 获取整个配置
const config = moduleConfigManager.get('my-module');

// 获取特定值
const timeout = moduleConfigManager.get('my-module', 'timeout');

// 获取嵌套值
const maxRetries = moduleConfigManager.get('my-module', 'retry.max');

// 带默认值
const value = moduleConfigManager.get('my-module', 'nonexistent', 'default');
```

### 4.4 更新配置（热更新）

```javascript
moduleConfigManager.set('my-module', { timeout: 60 });
```

### 4.5 监听配置变化

```javascript
moduleConfigManager.watch('my-module', (config) => {
  console.log('配置已更新:', config);
  // 应用新配置
});
```

### 4.6 环境变量覆盖

使用环境变量覆盖配置：

```bash
# 格式: MODULE_<模块名>_<配置键>
MODULE_MY_MODULE_TIMEOUT=60
MODULE_MY_MODULE_ENABLED=false
```

### 4.7 配置文件

在 `config/modules/my-module.json` 中创建配置文件：

```json
{
  "timeout": 60,
  "enabled": true,
  "retry": {
    "max": 5,
    "delay": 2000
  }
}
```

## 5. 错误处理

### 5.1 使用 asyncHandler 包装路由

```javascript
const { asyncHandler } = require('./middleware/module-error-handler');

router.get('/data', asyncHandler(async (req, res) => {
  // 异步代码中的异常会被自动捕获
  const data = await fetchData();
  res.json({ success: true, data });
}));
```

### 5.2 创建模块异常

```javascript
const { createModuleError } = require('./middleware/module-error-handler');

throw createModuleError('NOT_FOUND', '记录不存在', {
  id: 123,
  type: 'asset',
});
```

### 5.3 标准异常类型

| 类型 | HTTP状态 | 使用场景 |
|------|---------|---------|
| VALIDATION_ERROR | 400 | 参数验证失败 |
| NOT_FOUND | 404 | 记录不存在 |
| FORBIDDEN | 403 | 权限不足 |
| CONFLICT | 409 | 业务规则冲突 |
| DEPENDENCY_ERROR | 503 | 外部依赖失败 |
| INTERNAL_ERROR | 500 | 系统内部错误 |

### 5.4 安全执行

```javascript
const { safeExecute, safeExecuteWithRetry } = require('./middleware/module-error-handler');

// 安全执行（失败返回默认值）
const result = await safeExecute(
  () => fetchData(),
  defaultValue,
  '获取数据失败'
);

// 带重试的安全执行
const result = await safeExecuteWithRetry(
  () => callExternalAPI(),
  3,    // 最大重试次数
  1000, // 重试间隔（毫秒）
  null  // 默认值
);
```

## 7. 最佳实践

### 7.1 模块设计原则

1. **单一职责**: 每个模块只负责一个业务领域
2. **接口隔离**: 通过标准接口与外部通信
3. **依赖倒置**: 依赖抽象而非具体实现
4. **事件驱动**: 使用事件总线进行模块间通信
5. **错误隔离**: 单个模块异常不影响其他模块

### 7.2 命名规范

- 模块名称: 小写字母 + 连字符 (e.g., `quality-control`)
- 事件名称: 小写字母 + 连字符 (e.g., `asset-created`)
- 配置键: 小写字母 + 下划线 (e.g., `cache_ttl`)

### 7.3 目录结构

```
backend/
├── architecture/          # 架构规范
│   └── module-interfaces.js
├── adapters/              # 模块适配器
│   ├── asset-module-adapter.js
│   └── quality-control-module-adapter.js
├── config/
│   ├── module-registry.js
│   ├── module-config-manager.js
│   └── modules/           # 模块配置文件
│       ├── asset.json
│       └── quality-control.json
├── middleware/
│   └── module-error-handler.js
├── testing/
│   ├── module-test-framework.js
│   └── test-module-architecture.js
├── init-modules.js        # 模块注册初始化
└── server.js
```

### 7.4 迁移指南

从旧代码迁移到新架构：

1. 创建模块适配器类，实现标准接口
2. 将原有业务逻辑迁移到适配器中
3. 在 `init-modules.js` 中注册模块
4. 在路由中使用 `moduleRegistry.get()` 获取模块
5. 用 `asyncHandler` 包装路由处理函数
6. 用事件通信替代直接调用其他模块

## 附录

### A. 模块生命周期

```
注册 → 配置加载 → 初始化 → 启动 → 运行 → 停止 → 销毁
```

### B. 配置优先级

```
默认配置 < 环境变量 < 配置文件 < 运行时设置
```

### C. 事件流示例

```
资产状态变更
    ↓
asset:updated 事件
    ↓
质控模块监听
    ↓
检查是否需要质控
    ↓
quality-control:inspection-required 事件
    ↓
其他模块响应
```
