## 二、如何增加一个新模块

### 2.1 步骤概览

增加一个新模块只需 **4个步骤**：

```bash
步骤1: 创建模块目录结构
步骤2: 编写模块配置文件
步骤3: 编写业务代码（路由/服务）
步骤4: 重启服务（自动加载）
```

### 2.2 详细步骤

#### 步骤1: 创建模块目录

```bash
# 在 backend/modules/ 下创建新模块目录
mkdir -p backend/modules/my-new-module/{config,routes,services,models}
```

目录结构：
```
backend/modules/my-new-module/
├── config/
│   └── module.config.js      # 模块配置文件（必需）
├── routes/
│   └── index.js              # 路由入口（必需）
├── services/                 # 业务服务（可选）
│   └── my-service.js
└── models/                   # 数据模型（可选）
    └── my-model.js
```

#### 步骤2: 编写配置文件

**config/module.config.js**:

```javascript
module.exports = {
  // 模块基本信息
  id: 'my-new-module',                    // 模块唯一ID（必需）
  name: '我的新模块',                      // 模块显示名称
  version: '1.0.0',                       // 版本号
  description: '这是一个示例模块',         // 描述
  category: '业务管理',                    // 分类
  type: 'business',                       // 类型: system/business
  status: 'stable',                       // 状态: stable/beta/alpha
  
  // 依赖模块（可选）
  dependencies: [
    { 
      module_id: 'asset-management',      // 依赖的模块ID
      dependency_type: 'required',        // required 或 optional
      min_version: '1.0.0'                // 最低版本
    }
  ],
  
  // 功能开关配置
  features: {
    feature1: {
      enabled: true,                      // 默认启用
      name: '功能1',                       // 功能名称
      description: '这是功能1的描述',      // 功能描述
      configurable: true,                 // 是否可配置（用户可开关）
      default_enabled: true               // 默认开启状态
    },
    feature2: {
      enabled: true,
      name: '功能2',
      description: '这是功能2的描述',
      configurable: true,
      default_enabled: false               // 默认关闭
    }
  },
  
  // 后端配置
  backend_config: {
    api_prefix: '/api/my-new-module',     // API路由前缀（必需）
    routes_path: 'routes',                // 路由目录
    database_tables: ['my_table1', 'my_table2'],  // 使用的表
    services: [                           // 服务列表
      { name: 'MyService', path: 'services/my-service' }
    ]
  },
  
  // 前端配置（可选）
  frontend_config: {
    menu_keys: ['/my-module/page1', '/my-module/page2'],
    menu_routes: [
      {
        key: '/my-module',
        icon: 'SettingOutlined',          // Ant Design图标
        label: '我的模块',
        path: '/my-module',
        component: 'MyModulePage',
        permissions: ['my-module:read']
      }
    ],
    permissions: [
      'my-module:read',
      'my-module:create',
      'my-module:update',
      'my-module:delete'
    ]
  },
  
  // 配置项定义（用户可配置）
  config_schema: [
    {
      key: 'setting1',
      name: '设置项1',
      type: 'boolean',                    // boolean/number/string/select
      default: true,
      description: '这是设置项1的描述'
    },
    {
      key: 'setting2',
      name: '设置项2',
      type: 'number',
      default: 100,
      min: 0,
      max: 1000,
      description: '数值设置'
    },
    {
      key: 'setting3',
      name: '设置项3',
      type: 'select',
      options: [
        { label: '选项A', value: 'a' },
        { label: '选项B', value: 'b' }
      ],
      default: 'a'
    }
  ],
  
  default_config: {
    setting1: true,
    setting2: 100,
    setting3: 'a'
  }
};
```

#### 步骤3: 编写路由

**routes/index.js**:

```javascript
const express = require('express');
const router = express.Router();
const moduleConfig = require('../config/module.config');

// 检查功能是否启用的辅助函数
function isFeatureEnabled(featureName) {
  const feature = moduleConfig.features[featureName];
  return feature && feature.enabled;
}

// ========== 功能1的路由 ==========
if (isFeatureEnabled('feature1')) {
  
  // GET /api/my-new-module/items
  router.get('/items', async (req, res) => {
    try {
      // 获取租户ID
      const tenantId = req.user.tenant_id;
      
      // 查询数据库
      const db = require('../../config/database');
      const [rows] = await db.execute(
        'SELECT * FROM my_table1 WHERE tenant_id = ?',
        [tenantId]
      );
      
      res.json({ success: true, data: rows });
    } catch (error) {
      console.error('查询失败:', error);
      res.status(500).json({ success: false, message: '查询失败' });
    }
  });
  
  // POST /api/my-new-module/items
  router.post('/items', async (req, res) => {
    try {
      const tenantId = req.user.tenant_id;
      const { name, description } = req.body;
      
      const db = require('../../config/database');
      const [result] = await db.execute(
        'INSERT INTO my_table1 (tenant_id, name, description) VALUES (?, ?, ?)',
        [tenantId, name, description]
      );
      
      res.json({ 
        success: true, 
        message: '创建成功',
        data: { id: result.insertId }
      });
    } catch (error) {
      console.error('创建失败:', error);
      res.status(500).json({ success: false, message: '创建失败' });
    }
  });
}

// ========== 功能2的路由 ==========
if (isFeatureEnabled('feature2')) {
  
  router.get('/reports', async (req, res) => {
    // 功能2的实现
    res.json({ success: true, data: [] });
  });
}

// 模块状态接口（必需）
router.get('/status', (req, res) => {
  res.json({
    success: true,
    data: {
      module_id: moduleConfig.id,
      name: moduleConfig.name,
      version: moduleConfig.version,
      features: Object.entries(moduleConfig.features).map(([key, feature]) => ({
        id: key,
        name: feature.name,
        enabled: feature.enabled
      }))
    }
  });
});

// 模块配置接口（可选）
router.get('/config', (req, res) => {
  res.json({
    success: true,
    data: {
      schema: moduleConfig.config_schema,
      default: moduleConfig.default_config
    }
  });
});

module.exports = router;
```

#### 步骤4: 重启服务

## 三、模块加载机制

### 3.1 自动发现

```javascript
// backend/core/module-loader.js 核心逻辑

1. 扫描 backend/modules/ 目录
2. 读取每个模块的 config/module.config.js
3. 验证模块配置
4. 检查依赖关系
5. 加载路由文件
6. 注册API端点
7. 模块管理API（/api/system/modules）
```

### 3.2 加载流程

```
服务启动
    │
    ▼
扫描 modules/ 目录 ──► 发现所有模块
    │
    ▼
读取 module.config.js ──► 解析配置
    │
    ▼
验证配置 ──► 检查必需字段
    │
    ▼
检查依赖 ──► 确保依赖模块已加载
    │
    ▼
加载路由 ──► require(routes/index.js)
    │
    ▼
注册API ──► app.use(api_prefix, router)
    │
    ▼
完成加载 ──► 模块可用
```

#### 步骤3: 路由文件

```javascript
// backend/modules/equipment-loan/routes/index.js

const express = require('express');
const router = express.Router();
const moduleConfig = require('../config/module.config');

function isFeatureEnabled(featureName) {
  const feature = moduleConfig.features[featureName];
  return feature && feature.enabled;
}

// 借出管理功能
if (isFeatureEnabled('loan_management')) {
  
  // 获取借出列表
  router.get('/loans', async (req, res) => {
    const tenantId = req.user.tenant_id;
    const db = require('../../config/database');
    
    const [rows] = await db.execute(
      `SELECT l.*, a.asset_name, a.asset_code 
       FROM equipment_loans l
       JOIN assets a ON l.asset_id = a.id
       WHERE l.tenant_id = ?
       ORDER BY l.created_at DESC`,
      [tenantId]
    );
    
    res.json({ success: true, data: rows });
  });
  
  // 创建借出申请
  router.post('/loans', async (req, res) => {
    const tenantId = req.user.tenant_id;
    const { asset_id, borrower, borrow_date, expected_return_date, purpose } = req.body;
    
    const db = require('../../config/database');
    const [result] = await db.execute(
      `INSERT INTO equipment_loans 
       (tenant_id, asset_id, borrower, borrow_date, expected_return_date, purpose, status)
       VALUES (?, ?, ?, ?, ?, ?, 'pending')`,
      [tenantId, asset_id, borrower, borrow_date, expected_return_date, purpose]
    );
    
    res.json({ success: true, data: { id: result.insertId } });
  });
}

// 归还管理功能
if (isFeatureEnabled('return_management')) {
  
  // 归还设备
  router.post('/loans/:id/return', async (req, res) => {
    const { id } = req.params;
    const { return_date, condition, remarks } = req.body;
    
    const db = require('../../config/database');
    await db.execute(
      `UPDATE equipment_loans 
       SET status = 'returned', return_date = ?, condition = ?, remarks = ?
       WHERE id = ?`,
      [return_date, condition, remarks, id]
    );
    
    res.json({ success: true, message: '归还成功' });
  });
}

// 模块状态
router.get('/status', (req, res) => {
  res.json({
    success: true,
    data: {
      module_id: moduleConfig.id,
      name: moduleConfig.name,
      version: moduleConfig.version,
      features: Object.entries(moduleConfig.features).map(([key, feature]) => ({
        id: key,
        name: feature.name,
        enabled: feature.enabled
      }))
    }
  });
});

module.exports = router;
```

## 六、总结

### 增加模块的4个步骤：

1. **创建目录** - `mkdir -p backend/modules/模块名/{config,routes}`
2. **编写配置** - `config/module.config.js` 定义模块信息
3. **编写路由** - `routes/index.js` 实现业务API
4. **重启服务** - 自动加载新模块

### 关键要点：

- 每个模块是**独立的**，可以单独启用/禁用
- 模块内的功能也可以**单独开关**
- 模块可以定义**依赖关系**，系统会自动检查
- 租户可以**独立配置**每个模块的参数
- 模块API会自动注册，**无需修改主代码**
