- 可能导致资产状态不一致
- 没有检查是否有其他未完成的调配申请

**修复内容**:

```javascript

  });
}

// 检查是否有其他未完成的调配记录（在同一租户内）
const [otherTransfers] = await connection.execute(
  `SELECT id FROM transfer_records tr
   WHERE 1=1 ${tenantFilter.whereClause} AND tr.asset_code = ? AND tr.id != ? AND tr.status IN (?, ?)`,

  await connection.rollback();
```

### 8. ✅ 批量导入资产缺少事务

**严重程度**: 轻微

**位置**: `backend/routes/assets/asset.import-export.js` - `POST /import`

**问题描述**:
- 批量导入资产时，使用循环逐个插入，但没有使用事务
- 如果中间某个资产导入失败，前面的资产已经插入，无法回滚

**修复内容**:

```javascript
async function importHandler(req, res) {
  try {
    const uploadedFile = pickUploadedFile(req);
    if (!uploadedFile) {
      return res.status(400).json({ success: false, message: '请选择要上传的Excel文件' });
    }

    const fileBuffer = resolveUploadedFileBuffer(uploadedFile);
    const { headers, rows } = await parseImportRowsFromBuffer(fileBuffer);
    const { validRows, failedRows } = await preValidateImportRows(db, req, rows);

    const tenantId = getTenantId(req);
    let successCount = 0;
    const persistedFailedRows = [...failedRows];

    if (validRows.length > 0) {
      await withImportTransaction(async connection => {  // ✅ 使用事务
        for (const item of validRows) {
          try {
            await insertAsset(connection, tenantId, item.normalizedData);
            successCount += 1;
          } catch (error) {
            const persistError = resolveAssetPersistError(error);
            persistedFailedRows.push({
              rowNumber: item.rowNumber,
              error: persistError.message || '入库失败',
              rowData: item.rowData,
            });
          }
        }
      });
    }
    // ...
  } catch (error) {
    logger.error('导入资产失败:', error.message);
    return res.status(400).json({ success: false, message: error.message || '导入资产失败' });
  }
}

async function withImportTransaction(callback) {
  const connection = await db.getConnection();
  try {
    await connection.beginTransaction();  // ✅ 开始事务
    const result = await callback(connection);
    await connection.commit();  // ✅ 提交事务
    return result;
  } catch (error) {
    await connection.rollback();  // ✅ 回滚事务
    throw error;
  } finally {
    connection.release();
  }
}
```

### 9. ✅ 维修完成时缺少状态检查

**严重程度**: 轻微

**位置**: `backend/services/maintenance/requests.service.js` - `POST /requests/:id/complete`

**问题描述**:
- 完成维修时，没有检查资产当前状态是否为"维修中"
- 可能错误地更新了其他状态的资产

**修复内容**:

```javascript
async function completeRequest(id, body, req) {
  // ... 验证逻辑 ...

  const connection = await db.getConnection();
  try {
    await connection.beginTransaction();

    // ... 更新维修记录 ...

    const assetTenantFilter = addTenantFilter(req, 'a');
    const [assets] = await connection.execute(
      `SELECT a.* FROM assets a WHERE a.asset_code = ? ${assetTenantFilter.whereClause} FOR UPDATE`,  // ✅ 行锁
      [request.asset_code, ...assetTenantFilter.params],
    );

    if (assets.length === 0) {
      await connection.rollback();
      return {
        statusCode: 404,
        body: { success: false, message: '资产不存在' },
      };
    }

    const currentStatus = assets[0].status;
    const normalizedStatus = normalizeStatus(currentStatus, 'asset');
    
    // ✅ 检查资产状态
    if (normalizedStatus !== ASSET_STATUSES.MAINTENANCE && currentStatus !== '维修') {
      await connection.rollback();
      return {
        statusCode: 400,
        body: {
          success: false,
          message: `资产当前状态为"${currentStatus}"，无法完成维修。资产必须处于"维修中"或"维修"状态才能完成维修。`,
        },
      };
    }

    await connection.execute(
      `UPDATE assets a SET a.status = ?, a.updated_at = NOW()
       WHERE a.asset_code = ? ${assetTenantFilter.whereClause}`,
      [ASSET_STATUSES.IN_USE, request.asset_code, ...assetTenantFilter.params],
    );

    await connection.commit();

    return {
      statusCode: 200,
      body: { success: true, message: '维修已完成，已创建维护日志' },
    };
  } catch (error) {
    await connection.rollback();
    throw error;
  } finally {
    connection.release();
  }
}
```

## 🔧 修复技术细节

### 1. 事务管理

所有修复都遵循统一的事务管理模式：

```javascript
const connection = await db.getConnection();
try {
  await connection.beginTransaction();
  
  // 业务逻辑
  
  await connection.commit();
} catch (error) {
  await connection.rollback();
  throw error;
} finally {
  connection.release(); // 始终释放连接
}
```

### 2. 并发控制

关键操作使用 `SELECT ... FOR UPDATE` 行锁：

```javascript
const [assets] = await connection.execute(
  'SELECT * FROM assets WHERE id = ? FOR UPDATE',
  [asset_id]
);
```

### 3. 状态验证

状态变更前进行严格验证：

```javascript
if (assets[0].status !== '调配中') {
  await connection.rollback();
  connection.release();
  return res.status(400).json({
    success: false,
    message: `资产当前状态为"${assets[0].status}"，无法完成操作`
  });
}
```

### 4. 错误处理

所有错误路径都确保连接被正确释放：

```javascript
if (errorCondition) {
  await connection.rollback();
  connection.release(); // 在所有早期返回中都释放连接
  return res.status(400).json({ success: false, message: 'error' });
}
```
